Registrace postavy nefunguje, plus tabulková schémata zde: create table public.characters (
  id integer not null,
  user_id integer null,
  first_name character varying(100) null,
  middle_name character varying(100) null,
  last_name character varying(100) null,
  birth_date timestamp without time zone null,
  is_active boolean null,
  created_at timestamp without time zone null,
  updated_at timestamp without time zone null,
  school character varying(100) null,
  description text null,
  death_date timestamp without time zone null,
  death_reason text null,
  avatar text null,
  height integer null,
  weight integer null,
  height_set_at timestamp without time zone null,
  school_set_at timestamp without time zone null,
  residence text null,
  is_system boolean null,
  character_history text null,
  show_history_to_others boolean null,
  constraint characters_pkey primary key (id),
  constraint characters_user_id_fkey foreign KEY (user_id) references users (id)
) TABLESPACE pg_default;

create table public.users (
  id serial not null,
  username text not null,
  email text not null,
  password text not null,
  role text null default 'user'::text,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  is_banned boolean null default false,
  ban_reason text null,
  banned_at timestamp with time zone null,
  is_system boolean null default false,
  character_order text null,
  highlight_words text null,
  highlight_color text null,
  can_narrate boolean null default false,
  narrator_color text null,
  constraint users_pkey primary key (id),
  constraint users_email_key unique (email)
) TABLESPACE pg_default;

Zkontroluj aby se správně načítala databazáze, praocvaly správně logiky invites kodů create table public.invite_codes (
  id bigint generated always as identity not null,
  code text not null,
  is_used boolean null default false,
  created_at timestamp with time zone null default CURRENT_TIMESTAMP,
  updated_at timestamp with time zone null default CURRENT_TIMESTAMP,
  created_by_user_id bigint null,
  constraint invite_codes_pkey primary key (id),
  constraint invite_codes_code_key unique (code),
  constraint invite_codes_created_by_user_id_fkey foreign KEY (created_by_user_id) references users (id)
) TABLESPACE pg_default;
